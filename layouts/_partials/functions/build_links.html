"icon" $type "name" (humanize $type) ) -}} {{- end -}} {{- end -}} {{- end -}} {{- /* 3. 处理其他自定义 url_ 参数 */ -}} {{- range $key, $value := $page.Params -}} {{- if and (hasPrefix $key "url_") (not (in $url_params $key)) -}} {{- $type := strings.TrimPrefix $key "url_" -}} {{- $seenKey := printf "url-%s-%s" $type (md5 $value) -}} {{- if not (index $seen $seenKey) -}} {{- $seen = merge $seen (dict $seenKey true) -}} {{- $links = $links | append (dict "url" $value "icon" $type "name" (humanize $type) ) -}} {{- end -}} {{- end -}} {{- end -}} {{- /* 返回结果 */ -}} {{- return $links -}}
